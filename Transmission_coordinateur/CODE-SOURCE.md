### `scripts/documentation_reminder.py`
```python
#!/usr/bin/env python3
"""
Rappel automatique de documentation pour le d√©veloppeur.
Analyse les changements Git et rappelle la documentation obligatoire.
"""

import os
import sys
import subprocess
from datetime import datetime
from pathlib import Path

def get_git_changes():
    """R√©cup√®re les fichiers modifi√©s depuis le dernier commit"""
    try:
        result = subprocess.run(['git', 'diff', '--name-only', 'HEAD'], 
                              capture_output=True, text=True, check=True)
        return result.stdout.strip().split('\n') if result.stdout.strip() else []
    except:
        return []

def get_uncommitted_changes():
    """R√©cup√®re les fichiers non commit√©es"""
    try:
        result = subprocess.run(['git', 'status', '--porcelain'], 
                              capture_output=True, text=True, check=True)
        return result.stdout.strip().split('\n') if result.stdout.strip() else []
    except:
        return []

def check_documentation_needed():
    """D√©termine si de la documentation est n√©cessaire"""
    changes = get_git_changes()
    uncommitted = get_uncommitted_changes()
    
    all_changes = changes + uncommitted
    
    # Patterns qui n√©cessitent de la documentation
    doc_triggers = [
        '.py',    # Tout fichier Python
        'config', # Fichiers de configuration
        'README', # Documentation
        'yaml',   # Configuration YAML
        'json'    # Configuration JSON
    ]
    
    needs_doc = any(
        any(trigger in change.lower() for trigger in doc_triggers)
        for change in all_changes
        if change.strip()
    )
    
    return needs_doc, all_changes

def show_reminder():
    """Affiche le rappel de documentation"""
    needs_doc, changes = check_documentation_needed()
    
    print("üìù RAPPEL DOCUMENTATION OBLIGATOIRE")
    print("=" * 50)
    
    if needs_doc:
        print("‚ö†Ô∏è  DOCUMENTATION REQUISE !")
        print("\nüîÑ Changements d√©tect√©s:")
        for change in changes:
            if change.strip():
                print(f"   ‚Ä¢ {change.strip()}")
        
        print("\nüìã ACTIONS OBLIGATOIRES:")
        print("   1. ‚úÖ Documenter dans journal_developpement.md")
        print("      python luxa/scripts/doc-check.py --update")
        print("\n   2. ‚úÖ Mettre √† jour TaskManager")
        print("      task-master show 11  # Voir la t√¢che documentation")
        print("      task-master set-status --id=11.X --status=done")
        print("\n   3. ‚úÖ Commiter les changements")
        print("      git add .")
        print("      git commit -m 'feat: [description]'")
        
        print("\n‚ö° RACCOURCI RAPIDE:")
        print("   python luxa/scripts/doc-check.py --update && \\")
        print("   task-master show 11")
        
    else:
        print("‚úÖ Aucune documentation imm√©diate requise")
        print("üí° Continuez le d√©veloppement normalement")
    
    return needs_doc

def main():
    """Fonction principale"""
    print("üîç Analyse des changements en cours...")
    
    if show_reminder():
        print("\nüö® IMPORTANT: Documentez avant de continuer le d√©veloppement!")
        sys.exit(1)  # Force la documentation
    else:
        print("\nüéØ Vous pouvez continuer le d√©veloppement")
        sys.exit(0)

if __name__ == "__main__":
    main()
```
